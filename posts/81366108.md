1、OC->JS 传数组的方法：
NSString \*arrStr = [self.arr componentsJoinedByString:@","];
NSString \*js = [NSString stringWithFormat:@"text('%ld',[%@]);",index,arrStr];
[self.wkView evaluateJavaScript:js
                  completionHandler:^(id \_Nullable parm1, NSError \* \_Nullable error) {
                      NSLog(@"%@",[error description]);
}];

2、OC->JS 传字典的方法：
 NSString \*jscript = [NSString stringWithFormat:@"post('%@',{%@});", self.URLString, self.postData];
    // 调用JS代码

}];
