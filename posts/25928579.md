#include "stdio.h"
#include "stdlib.h"
#include "string.h"


struct DATA       //定义链表结构
{
int id;//数据
struct DATA \*next;//指针
};


void linkAppend(struct DATA \*\*head, int id)   //链表的添加函数
{
struct DATA \*temp, \*node;
node = (struct DATA\*)malloc(sizeof(struct DATA));//  为结点分配内存空间
node->id = id;
node->next = NULL;     //令指向下一个结点的指针指向空。


if(\*head == NULL)//新的结点为第一个结点
{
\*head = node;
return;
}
else
{
temp = \*head;      
while(temp->next != NULL)//链表已有若干结点，找到最后一个结点
{
temp = temp->next;  
}
temp->next = node;   // 由上循环找到最后一个结点，让当前最后的结点指向这个新的结点。
}
return;
}


void linkShow(struct DATA \*head)//   打印输出函数
{
struct DATA \*temp;          // 设置结构体中间指针变量
if(head == NULL)
{
perror("该链表目前没有结点");
return;
}
temp = head;
while(temp != NULL)
{
printf("id=%d\n", temp->id); //  循环输出结点的数据
temp = temp->next;    // 依次指向下一个结点
}
}
void linkFree(struct DATA \*head)    // 链表空间释放函数
{
struct DATA \*temp;
while(head != NULL)
{
temp = head->next;//  释放前必须保存这个结点的指向
free(head);    //   释放当前结点
head = temp;    //  指向下一个结点
printf("sdkldj\n");
}
}


void linkInsert(struct DATA \*head, int id, int n)      / /   链表插入函数
{
struct DATA \*temp, \*node;
int cnt = 0;  //  做一个计数器，找到要插入结点的位置


node = (struct DATA \*)malloc(sizeof(struct DATA));
node->id = id;
node->next = NULL;


temp = head;
if(n == 0) //新结点要为头结点
{
node->next = head;        
head = node;
return;
}


while(temp->next != NULL)    //  不插入到头结点的时候
{
cnt ++;
if( cnt == n - 1)    
{
break;
}
temp = temp->next;   //否则继续查找要插入的位置
}
if(temp->next == NULL)//新结点插入在最后
temp->next = node;
else                     / /新结点插入在链表中间(temp, temp->next之间)
{
node->next = temp->next;
temp->next = node;
}
}
int linkCount(struct DATA \*head)   //  链表计数器函数。  
{
struct DATA \*temp;
int cnt = 0;
temp = head;
while(temp != NULL)
{
cnt ++;
temp = temp->next;
}
return cnt;  
}

void linkDel(struct DATA \*\*head, int n)// 链表删除第  n 个结点函数,  
{
struct DATA \*temp, \*pre;
temp = \*head;
pre = \*head;
int cnt = 0;
if(n <= 1)
{
\*head = (\*head)->next;
free(temp);  //   删除释放结点
temp = NULL;
return;
}
while(temp != NULL)
{
if(cnt == n - 1)     //   查找要删除的结点
{
pre->next = temp->next;
free(temp);
temp = NULL;
return;
}
cnt ++;
if(temp->next == NULL)// 为后继的if(temp->next == NULL)准备
break;
pre = temp;
temp = temp->next;
}


if(temp->next == NULL)
{
pre->next = NULL;
free(temp);
temp = NULL;
return;
}
}

{
struct DATA \*head = NULL;//指向链表第一个节点

}
