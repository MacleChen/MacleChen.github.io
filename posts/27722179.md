今日结果：
       1  ps  -ef  | grep bash   |  grep  bin      // 可以加多个管道，进行过滤
            grep  main   main.c  //  过滤出main.c   文件中的main字符串的行
           grep  [options]    文件名    
                         -i   不分大小写
                        -n  所在的行数
                        -q  与其他命令使用，抑制显示
            grep  -ni  main  main.c  | wc  -l       过滤出个数
                        -v  反向匹配
                        -l  列出要查找字符串的所在文件
        2  基本的特殊字符集：    ^   最左边开始匹配                grep  ^main    main.c
                                                grep   'argv)'    main.c    最右边开始匹配
                                               grep  ^$  main.c    查看空行                  （进入编辑框时，:set nu  设置行数）
                                              grep  '\<wo' main.c   在字的开始进行匹配
                                              grep  'd\>'  main.c     在字的结尾进行匹配
                                              grep  'w.o'  main.c    对任何单个字符进行匹配
                                              grep  ['who'] main.c  对多个字符进行匹配
                                              grep  [^str]   main.c  对任何不在str 字符中进行匹配
                                              grep  '\\'   main.c    查找 '\'  字符
        3  find 命令：    find ./  -name  '\*.c '   找当前目录下的.c 文件（包括子目录）
                                find ./ -group  user1  查找改组的文件
                                -type  文件类型查找          find ./ -type  d    查找目录文件  （f   查找普通文件）
                                -user  通过用户查找
                                -exec  （很有用）  查出文件并且可以加执行命令
                               find  ./  -group  root  -exec echo  {}  \;     //  {} 执行结果集      \;  结束标识
                               find ./  -name  '\*.c'   -exec  rm  -f {} \;   
       4  服务器:  ifconfig  eth0   // 找到配置文件           位置etc/sysconfig/network-scripts/    (16进制的可读性比较好）
                         ifdown eth0     // 关闭网卡             ifup  eth1  // 激活网卡
                         ifconfig  eth0   [options]         ip\_address          修改ip地址
                          interface     网络接口
                         ifconfig eth0  down  || up
                        broadcast address :   设置广播地址
                         ifconfig  eth1  broadcast 192.168.1.254
                         pointtopoint    :  启动点对点方式
                         netmask ：  修改子网掩吗       //  ifconfig eth1  netmask  255.255.255.128 
       5   ifconfig  eth0  查看eth0 的配置             
            netstat   查看网络状态信息            netstat  -an  | grep  23   查看23 号端口信息
            passwd  [用户名]   修改用户密码
