今日结果：
      1  linux 命令行的一般格式： 命令名  [选项]  参数
                ls  文件名    // 查看文件    ls  -l   // 查看各个文件的属性信息     ls  -lt || -tl  // 有次序查看
                ls -a  文件名  //  查看隐藏文件    ls  -ld  // 查看目录的属性  
                ls  -s  ...   // 查看每个文件的块大小      ls  --sort time  \* //  按时间排序当前文件，  (  ls --sort  size  -l  \*) 
      2  whatis ：  简单的描述命令，     whatis ls  // 当前目录文件下的简单查看。    
      3  man  命令说明（常用）     man ls   ；   （可以查看c/ c++库函数）   info  更详细的文件说明
      4  cd  ..  切换上一级目录     cd .   当前目录     cd  切换根目录
      5  mkdir  temp   // 创建目录    rmdir // 删除目录（目录必须为空）
      6  cp  source     destination   //  复制文件        cp   a.out   b.out  
              rm   -rf  temp     // 删除文件夹   （  强制删除，不询问）
              rm    -r   tmep  //  删除文件夹（询问  y / n）    rm    a.out   // 删除文件
              mv   a.out  temp   //  移动文件a.out 到  temp 中     mv  -rf  \*   temp    将当前的所有文件都移动到temp中。
      7  drwxr-xr-r       //   d (表示目录文件） r (4 可读）  w (2可写）  x(1可执行）  
            rename    重命名     rename  abc   def   abc.\*   // 把abc.\* 的文件名用def 代替
            find   ./   -name  ' \*.c'  //  查找当前目录下的.c 文件
            more  分屏显示文件内容   
            less  与more 很像 但还具有更多的功能
            tail （很有用） 查看文件末尾（默认10行）    tail  -20  main.c  //  查看末尾20 行
            wc   统计文件的行数，字数，  字节数     wc  main.c   
      8  加载环境变量（就是向系统添加命令）
      9  ps   查看进程    （可执行文件的运行状态）
            ps    -ef   //             kill   送信号给进程    kill  -9   19777    // 9（表示千杀） 19777（进程id) 表示强杀该进程
            kill   -l  //  列出所有的信号  
            pstree  //   树形显示进程        chkconfig    // 查看服务        ntsysv  //图形化设置启动/不启动服务
            top  // 工具       zombie   僵尸进程（子类进程结束后，父类进程还继续存在，保存子类的状态）
      10  df 查看磁盘空间    df  -mv  // 1M 为单位    df  -kv  // 1K为单位
              挂载文件或目录 ：   mount  -vfs     /mnt/hgfs/c   /etc/aa    // v(虚拟） f（文件） s（系统）    source   destination
                  注意： 挂载点的目录必须存在
              挂载镜像文件：   mount  -o  loop  /mnt/hafs/hard/文件名.iso    /dev/iso  
               接触挂载:    umount  /dev/iso  
              free  内存使用情况查询 
              cat  /proc/cpuinfo  查看系统配置信息
             cat  /proc/meminfo  查看内存配置信息
              date 日期查询  （有用）     date +%d  ||  +%y  || +%m        查看年，月，日
              bc  计算机 （quit退出)  ； clear   清楚terminal屏内容 ； who  查看有那些用户；  whoami 查看自己
