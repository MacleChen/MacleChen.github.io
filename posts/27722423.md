                 ： 命令模式  
                         ctrl +b  向前翻页              ctrl+f  向后翻页                     ctrl+d 半页向后翻                        O 移动光标到行首
                        shift+ins  粘贴       U  回复动作      r 代替字符   R 代替连续字符串          ESC 退出模式

       4  gcc 编译器：   -o  产生目标代码或可执行代码
                                  -E  只运行C预编译
                                  -C 生成中间代码 
                                  -S 生成汇编语言代码
                      一步执行C代码：  gcc  main.c -o main.out
                      两步执行C代码：      gcc  -c main.c       生成中间代码 main.o
                                                       gcc  main.o  -o  main   生成执行文件
                                  -g 调试程序
         5  运行多的.c 和 .h 文件
                   fun.c             fun.h       main.c
                  一边自己写的程序放在自己用户目录下 (inc 放置头文件）
              cc  或 gcc  ：      gcc \*c  -o  main   -I./inc    运行.h 文件夹       //生成中间代码
                                         gcc  -c  main.c  -o  main.o  -I./inc            //  生成可执行文件
                    gcc  \*.c  -o  main     生成可执行文件
         6   gdb(很重要）程序调试
                          gcc  \*.c  -I./inc  -g -o  main         加 -g 文件可以被调试
                           gdb  main   调试代码  
                              list   按 L 列出源代码  
                              break （b  8)  给第八行设置断点
                              run  ( r ) 运行  
                               continue   (   c  )   继续程序运行，直到下一个断点
                             step   ( s )  单步执行， 也进入函数单步执行
                             next  ( n ) 单步执行 ， 不进入函数体
                            print (  p  j )   查看该变量的值
