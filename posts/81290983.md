let userDictArray = NSMutableArray.init()

        for user in recentArrayTemp {

            // 模型转字典

            var userDict = user.mapJSON().dictionaryObject

            if userDict != nil {

                userDict?.removeValue(forKey: "merchant")  // ? 不能出现"<null>"不然会存储不到 UserDefaults

                userDictArray.add(userDict as Any)

            }

        }

UserDefaults.standard.set(NSArray.init(array: userDictArray), forKey: DICT\_USER\_RECENT)

UserDefaults.standard.synchronize()
