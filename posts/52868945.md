//将本地项目提交到coding上托管
方法：
1. git init
// 增加git管理
2. git clone <目标远程地址>  // 之后当前目录会出现 远程上对应的目录  （克隆远程目标地址的内容）
3. cd <远程目标地址的根目录>
5. git pull <coding上拷贝的远程目标地址（https SSH）>    
6. git add . // 加入项目
// 也可增加文件夹（例如：mkdir test1,并向文件夹中添加README.md文件帮助文件夹上传成功）
7. git commit -m “项目提交描述信息”  //  提交项目到本地库
8. git remote add  origin <coding上拷贝的目标远程地址>  
9. git push -u origin master  // 将项目推送到目标远程地址
