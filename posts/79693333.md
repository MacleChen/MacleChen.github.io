#### 1. 在mac系统下安装虚拟环境：（以下都是终端命令, 有的pip命令用不了，请使用pip3）

pip install virtualenv         // 记得不要sudo ，因为会安装到系统目录下

pip install virtualenvwrapper // 安装虚拟环境管理器（没有这个就不能是用rmvirtualenv 去删除虚拟环境）

#### 2. 配置 virtualenvwrapper 管理器

##### 1》打开.brash\_profile文件

vim ~/.bash\_profile

##### 2》编辑文件（在文件末尾编写以下内容）

export WORKON\_HOME=~/py\_envs        # 加入自己设定虚拟环境工作空间（第三步会创建这个目录）
export VIRTUALENVWRAPPER\_PYTHON=/Library/Frameworks/Python.framework/Versions/3.6/bin/python3  # 设置默认的python版本（可以有些人不太一样，那使用命令查找：which python或which python3）

source /Library/Frameworks/Python.framework/Versions/3.6/bin/virtualenvwrapper.sh   # 添加要激活的虚拟环境管理器的路径（如果不太清楚路径，请用命令查找：sudo find / -name virtualenvwrapper.sh）

##### 3》生效或激活虚拟环境管理器的相关命令

source ~/.bash\_profile

#### 3. 创建虚拟环境目录

mkdir ~/py\_envs                    // 创建虚拟环境目录

cd ~/py\_envs                        // 进入虚拟环境目录中

virtualenv  env\_workspace001    // 创建虚拟环境

#### 4. 激活虚拟环境

cd env\_workspace001        // 进入虚拟环境

source ./bin/activate         // 激活虚拟环境

#### 5. 使用虚拟环境(安装虚拟环境中的插件，还是不要使用sudo)

pip install flask

pip install django

pip install scipy

#### 6. 退出虚拟环境

deactivate

#### 7. 删除环境（退出虚拟环境，并退出虚拟环境所在的虚拟路径）

rmvirtualenv  虚拟环境名称

#### 8. 其它方法

mkvirtualenv py\_db        // 通过wrapper创建工作空间

workon  环境名称            // 切换虚拟环境

lsvirtualenv -b            // 列出所有的虚拟环境

virtualenvwrapper  -help        // 查看所有命令
