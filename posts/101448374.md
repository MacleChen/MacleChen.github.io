1. 运行在Android上：

       react-native run-android      // 默认运行设备

       react-native run-android --simulator="XXX"    // 模拟器运行

       react-native run-android --device="chenfan\_mate20pro"   // 真机（chenfan\_mate20pro 设置真机的设备名称）前提：必须连接上ios手机设备

2. 运行在iOS上：

       react-native run-ios      // 默认运行设备

       react-native run-ios --simulator="iPhone XR"    // 模拟器运行

       react-native run-ios --device="chenfaniPhone"   // 真机（chenfaniPhone 设置真机的设备名称）前提：必须连接上ios手机设备
