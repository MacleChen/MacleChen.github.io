                makefile   文件中：
%.o : %.cpp
g++  -c  $<   -o   $@      // $< 表示所有.cpp (%.cpp)    $@表示生成一个目标文件
  $^    表示所有的依赖文件
  ${CPP}: ${OBJS}
g++  S^  -o  S@   ${CPP}

              make  -f  tt     //  将makefile文件改名为tt进行编译
