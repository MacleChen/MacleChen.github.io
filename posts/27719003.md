　　4  数据库完整性主要分为：实体完整性，参照完整性，用户自定义完整性； 实体完整性是：要么有一行记录，要么没有，不能有半行的；  UNIQUE (唯一约束）例如学号不能有重复，好PRIMARY(主键)约束相似。
       5  sqite> create table stu (id integer , name text);  // 创建表和字段属性

           sqite> .table //查询当前有哪些表
           sqite> attach database 'mybak.dat' as dbbak;     // 创建数据库    存储在当前目录下的mybak.dat中， 令别名 dbbak
           sqite> .help // 查看帮助
           sqite> .database  // 查看数据库目录            detach  dback ; // 分离数据库 （与U盘退出相似）
           sqite> .quit // 退出命令行       .schema  // 查看表的结构 （所有的）   .schema  stu (一张表）
           sqite> .output ./aaa.log  //重定向（将下次所查到的信息写入当前目录下的aaa.log中
           sqite> .output  c:/aaa.log  // 输入到指定的文件中。      .output  stdout   // 输出到DOS屏上
           sqite> .separator  \*\*  // 以\*\*将字段间的数据隔开
　　　sqite> .mode column  // 列的查询模式        .mode insert // 插入模式查询  
          sqite> .header on  //  查询时显示字段名称      .dump  dback;  // 查询表的结构      .delete from stu  ;删除表中的记录（但表壳任然存在）          .drop table stu ;  //  删除整张表（包括表壳）
